# ドメインモデル図

## 全体図
```uml
@startuml

scale 1

package {
  object パーティー {
      活動期間
  }

  object 参加申請 {
      状態（承認・未承認）
  }

  object 制作アイデア {
      テーマ
      制作物の種類
      説明
  }

  object メンバー募集 {
      募集ステータス
      ギルドメンバーID
  }

  object 募集役割 {
      役割名
      参考ジョブ
      参考スキル
      備考
  }
}

object 管理者
object 申請主
object パーティーメンバー

object ギルド
object パーティー申請書 {
  申請者ID
  活動期間
  制作アイデア
  募集役割リスト
}


package {
  object ギルドメンバー {
      学籍番号
      名前
      コース
      性別
      制作アイデアリスト
  }

  object 所持スキル {
      スキルレベル
      累計経験値
  }
}
object スキル {
    スキル名
}

package {
  object ジョブ {
      ジョブ名
      イメージ画像
  }
  object 取得条件 {
      必要レベル
  }
}

object 参加申請 {
    状態（承認・未承認）
}

object 参加パーティー


object 評価 {
    評価点
}

object 学生
object 担当教員
object 担当授業カリキュラム
object 授業カリキュラム {
    教科
    担当教員
    担当クラス
    期間
    実施曜日
    実施時間
}
object 教員

package {
  object 授業 {
      日時？
  }
  object 教科 {
      教科名
  }
  object 習得できるスキル
}

パーティー --> 管理者
管理者 --|> ギルドメンバー
申請主 --|> ギルドメンバー
パーティー --> 制作アイデア
パーティー o--> 募集役割
パーティー o--> 参加申請
パーティー o--> パーティーメンバー
パーティーメンバー --|> ギルドメンバー
参加申請 --> 申請主
募集役割 o--> メンバー募集
募集役割 --> スキル
募集役割 --> ジョブ

ギルドメンバー o--> 所持スキル
ギルドメンバー o--> ジョブ
ギルドメンバー o--> 参加パーティー
ギルドメンバー o--> 参加申請
所持スキル --> スキル

ジョブ o--> 取得条件
取得条件 --> スキル

授業 <-- 評価
評価 --> 学生
授業 --> 担当教員
授業 --> 教科
担当教員 --|> 教員
学生 --|> ギルドメンバー
教員 o--> 担当授業カリキュラム
担当授業カリキュラム --|> 授業カリキュラム
教科 o--> 習得できるスキル

ギルド o--> パーティー
ギルド o--> パーティー申請書
ギルドメンバー o--> パーティー申請書

@enduml
```
